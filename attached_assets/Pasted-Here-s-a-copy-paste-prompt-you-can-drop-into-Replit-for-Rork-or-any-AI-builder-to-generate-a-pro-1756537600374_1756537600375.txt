Build a Production-Ready Website for The Pressure Play

Data Models (TypeScript)

Create types/interfaces for:

type PlatformLink = { label: 'Spotify'|'Apple'|'YouTube'|'Google'|'RSS'|'Website'; url: string };
type TopicTag = { slug: string; title: string };

interface Guest {
  _id: string; name: string; role?: string; bio?: string;
  headshot?: string; socials?: Record<string,string>;
  tags?: string[]; episodeSlugs?: string[];
}
interface Episode {
  _id: string; number?: number; slug: string; title: string;
  description: string; cover?: string; duration?: number;
  publishedAt: string; platforms?: PlatformLink[];
  guests?: string[]; topics?: TopicTag[]; transcript?: any;
  highlights?: { t: string; text: string }[];
  youtubeId?: string; audioUrl?: string;
}
interface ContactMessage { name: string; email: string; subject?: string; message: string; }
